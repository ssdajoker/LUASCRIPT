```javascript
// Transpiled to Lua
(function()
    local __obj = obj
    if __obj ~= nil then
        local __prop = __obj["prop"]
        if __prop ~= nil then
            local __method = __prop["method"]
            if __method ~= nil and type(__method) == "function" then
                return __method()
            end
        end
    end
    return nil
end)()
```
